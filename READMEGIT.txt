$ git commit --amend -m 'update'  — изменить комментарий к последнему коммиту (локальная ветка)

$ git reset --hard HEAD~   — удалить последний коммит (локальная ветка)
$ git reset --hard HEAD~3  — удалить последние 3 (локальная ветка)
$git reset --hard HEAD@{1} — отменить последнее действие (локальная ветка)

git reset --hard 8809e — откат к конкретному коммиту (локальная ветка)
